Skip Line Number 5

This script reads all content from 'test.txt' and writes it into 'new_file.txt', skipping the 5th line. 

# Open the input file 'test.txt' in read mode
with open("test.txt", "r") as fp:
    # Read all lines from the input file
    lines = fp.readlines()

# Open the output file 'new_file.txt' in write mode
with open("new_file.txt", "w") as fp:
    count = 0
    # Iterate through each line in the input file
    for line in lines:
        # Skip the 5th line (index 4)
        if count == 4:
            count += 1
            continue
        else:
            # Write the current line to the output file
            fp.write(line)
        # Increment the count after processing each line
        count += 1
