Exercise 5: Create Dictionary by Extracting Keys

sampleDict = { 
  "name": "Kelly",
  "age": 25, 
  "salary": 8000, 
  "city": "New york" 
}

keys = ["name", "salary"]

# Solution 1: Dictionary Comprehension
newDict = {k: sampleDict[k] for k in keys}
print(newDict)

# Solution 2: Using update() method and loop
res = dict()
for k in keys:
    res.update({k: sampleDict[k]})
print(res)
