Exercise 6: Delete a List of Keys from a Dictionary

sample_dict = {
    "name": "Kelly",
    "age": 25,
    "salary": 8000,
    "city": "New york"
}
# Keys to remove
keys = ["name", "salary"]

# Solution 1: Using the pop() method and loop
for k in keys:
    sample_dict.pop(k)
print(sample_dict)

# Solution 2: Dictionary Comprehension
sample_dict = {
    "name": "Kelly",
    "age": 25,
    "salary": 8000,
    "city": "New york"
}
sample_dict = {k: sample_dict[k] for k in sample_dict.keys() - keys}
print(sample_dict)
